🎭 บทที่ 2: Playwright (UI Automation)

บทนี้จะสอนการใช้ Playwright ควบคุม browser อัตโนมัติ ตั้งแต่แนวคิดจนถึงเขียน test จริง

🤖 Playwright คืออะไร? มันคือ Tool ตัวไหน?

Playwright คือ library/tool ที่ Microsoft สร้างขึ้นมาเพื่อทำ automation โดยเฉพาะ ใช้ได้ทั้ง UI (ควบคุม browser) และ API (ยิง HTTP request)

ส่วนประกอบคืออะไรเปรียบเหมือน
VS Codeโปรแกรมที่ใช้เขียน codeโรงงาน (ที่ทำงาน)
Pythonภาษาที่ใช้เขียน scriptภาษาที่ใช้สั่งงาน
Playwrighttool ที่ใช้ทำ automationเครื่องจักรในโรงงาน
pytestตัวรัน test และสรุปผล pass/failผู้จัดการที่คอยสรุปผล
ทั้ง 4 อย่างนี้ต้องใช้ร่วมกันเสมอ ขาดอันใดอันนึงไม่ได้

VS Code → เปิดขึ้นมาเพื่อเขียน code
Python → ภาษาที่เราเขียน
Playwright → library ที่ import มาใช้ใน code
pytest → คำสั่งที่ใช้รัน test ใน Terminal

🆚 Playwright ทำ UI กับ API ต่างกันยังไง?

UI AutomationAPI Automation
ทำอะไรควบคุม browser กดปุ่ม กรอกฟอร์มยิง HTTP request โดยตรง ไม่เปิด browser
ใช้เมื่อทดสอบหน้าเว็บที่ user เห็นทดสอบ API endpoint
คำสั่งหลักpage.goto(), page.click(), page.fill()page.request.get(), page.request.post()
เร็วแค่ไหนช้ากว่า เพราะต้องเปิด browserเร็วกว่า ไม่ต้องเปิด browser

⌨️ คำสั่งเฉพาะของ Playwright ที่ต้องรู้

Playwright มีคำสั่งของตัวเองแบ่งเป็น 3 กลุ่มหลัก

กลุ่มที่ 1: Navigation — เปิดเว็บ ไปหน้าไหน

page.goto("https://www.saucedemo.com")  # เปิด URL
page.go_back()                           # กดปุ่ม back
page.go_forward()                        # กดปุ่ม forward
page.reload()                            # refresh หน้า

กลุ่มที่ 2: Action — ทำอะไรบนหน้าเว็บ

page.click("#login-button")              # กดปุ่ม
page.fill("#user-name", "standard_user") # กรอกข้อความ
page.select_option("#dropdown", "value") # เลือก dropdown
page.check("#checkbox")                  # ติ๊ก checkbox
page.uncheck("#checkbox")                # เอาติ๊กออก
page.screenshot(path="screenshot.png")  # ถ่ายรูปหน้าจอ
page.hover("#menu")                      # เอาเมาส์ชี้ที่ element

กลุ่มที่ 3: Assertion — เช็คผลว่าถูกต้องมั้ย

expect(element).to_be_visible()          # เช็คว่า element แสดงอยู่มั้ย
expect(element).to_be_hidden()           # เช็คว่า element ซ่อนอยู่มั้ย
expect(element).to_be_enabled()          # เช็คว่าปุ่มกดได้มั้ย
expect(element).to_contain_text("Login") # เช็คว่ามีข้อความนี้มั้ย
expect(page).to_have_url("https://...")  # เช็คว่า URL ถูกต้องมั้ย
expect(page).to_have_title("หน้าหลัก")  # เช็คว่า title ถูกต้องมั้ย
จำง่ายๆ:
Navigation = ไปไหน
Action = ทำอะไร
Assertion = เช็คว่าผลถูกมั้ย

🚀 เริ่มต้น Step 0 ถึงจบ — API Automation ด้วย Playwright

from playwright.sync_api import Playwright   # import Playwright

def test_get_user(playwright: Playwright):
    # สร้าง API request context — ไม่เปิด browser เลย
    api = playwright.request.new_context(base_url="https://jsonplaceholder.typicode.com")

    # ส่ง GET request
    response = api.get("/users/1")

    # เช็คผล
    assert response.status == 200
    body = response.json()
    assert "name" in body
    assert "email" in body

def test_create_post(playwright: Playwright):
    api = playwright.request.new_context(base_url="https://jsonplaceholder.typicode.com")

    # ส่ง POST request พร้อมข้อมูล
    response = api.post("/posts", data={
        "title": "test",
        "body": "hello",
        "userId": 1
    })

    assert response.status == 201

📄 สร้างไฟล์ pages/login_page.py

ไฟล์นี้คือ "แผนที่" ของหน้า login บอกว่า element ต่างๆ อยู่ที่ไหน และทำอะไรได้บ้าง

# ============================================================
# pages/login_page.py
# ไฟล์นี้ทำหน้าที่เดียวคือ บอกว่าหน้า login มีอะไรอยู่ที่ไหน
# ============================================================

class LoginPage:

    # __init__ คือ constructor = รันทันทีตอนสร้าง LoginPage ขึ้นมา
    # self = ตัวแทนของ object นี้เอง (เหมือน "ตัวเอง" ในภาษาไทย)
    # page = browser page ที่ Playwright ส่งมาให้
    def __init__(self, page):
        self.page = page

        # locator = วิธีบอก Playwright ว่า element อยู่ที่ไหนบนหน้าเว็บ
        # "#user-name" = หา element ที่มี id="user-name"
        # วิธีหา id: เปิดเว็บ → คลิกขวา → Inspect → ดู HTML
        self.username_input = page.locator("#user-name")

        # "#password" = หา element ที่มี id="password"
        self.password_input = page.locator("#password")

        # "#login-button" = หาปุ่ม login
        self.login_button = page.locator("#login-button")

        # "[data-test='error']" = หากล่องข้อความ error
        # ที่จะโชว์ตอน login ผิด
        self.error_message = page.locator("[data-test='error']")

    # method สำหรับเปิดหน้าเว็บ
    def goto(self):
        # สั่งให้ browser เปิด URL นี้
        self.page.goto("https://www.saucedemo.com")

    # method สำหรับกรอก username, password แล้วกด login
    # รับค่า username และ password มาเป็น parameter
    def login(self, username, password):
        # fill = กรอกข้อความลงใน input field
        self.username_input.fill(username)
        self.password_input.fill(password)

        # click = กดปุ่ม
        self.login_button.click()

📄 สร้างไฟล์ tests/test_login.py

ไฟล์นี้คือ test case จริงๆ บอกว่าจะทดสอบอะไร และคาดหวังผลอะไร

# ============================================================
# tests/test_login.py
# ============================================================

# ===== Import library =====
import pytest

# Page = type ของ browser page จาก Playwright
# expect = ตัวช่วยเช็คผลแบบ Playwright
from playwright.sync_api import Page, expect

# นำ LoginPage ที่เราสร้างไว้มาใช้
from pages.login_page import LoginPage


# ============================================================
# Test Cases
# ทุก function ที่ขึ้นต้นด้วย test_ จะถูก pytest รันเป็น test case
# page: Page = Playwright จะสร้าง browser page ให้อัตโนมัติ
# ============================================================


# ----------------------------------------------------------
# Test 1: login ด้วยข้อมูลถูกต้อง → ต้องเข้าหน้า inventory ได้
# ----------------------------------------------------------
def test_login_success(page: Page):
    # สร้าง LoginPage object โดยส่ง page เข้าไป
    login = LoginPage(page)

    # เปิดหน้าเว็บ
    login.goto()

    # กรอก username/password ที่ถูกต้อง แล้วกด login
    login.login("standard_user", "secret_sauce")

    # expect = เช็คว่าผลเป็นไปตามที่คาดมั้ย
    # to_have_url = URL ต้องเปลี่ยนไปเป็นหน้า inventory
    # ถ้าไม่ตรง → test fail ทันที พร้อมแสดง error message
    expect(page).to_have_url("https://www.saucedemo.com/inventory.html")


# ----------------------------------------------------------
# Test 2: login ด้วย password ผิด → ต้องเห็น error message
# ----------------------------------------------------------
def test_login_wrong_password(page: Page):
    login = LoginPage(page)
    login.goto()

    # ใส่ password ผิดตั้งใจ เพื่อทดสอบว่า error แสดงถูกมั้ย
    login.login("standard_user", "wrong_password")

    # to_be_visible = เช็คว่า element แสดงอยู่บนหน้าจอมั้ย
    expect(login.error_message).to_be_visible()

    # to_contain_text = เช็คว่ามีข้อความนี้อยู่มั้ย
    expect(login.error_message).to_contain_text("Username and password do not match")


# ----------------------------------------------------------
# Test 3: login โดยไม่กรอก username → ต้องเห็น error
# ----------------------------------------------------------
def test_login_empty_username(page: Page):
    login = LoginPage(page)
    login.goto()

    # ส่ง username เป็นค่าว่าง "" เพื่อทดสอบ validation
    login.login("", "secret_sauce")

    expect(login.error_message).to_contain_text("Username is required")


# ----------------------------------------------------------
# Test 4: login โดยไม่กรอก password → ต้องเห็น error
# ----------------------------------------------------------
def test_login_empty_password(page: Page):
    login = LoginPage(page)
    login.goto()

    # ส่ง password เป็นค่าว่าง ""
    login.login("standard_user", "")

    expect(login.error_message).to_contain_text("Password is required")


# ----------------------------------------------------------
# Test 5: login ด้วย account ที่ถูก lock → ต้องเห็น error
# ----------------------------------------------------------
def test_login_locked_user(page: Page):
    login = LoginPage(page)
    login.goto()

    # "locked_out_user" คือ account ที่ถูก lock ใน saucedemo
    login.login("locked_out_user", "secret_sauce")

    expect(login.error_message).to_contain_text("Sorry, this user has been locked out")

🔍 Locator คืออะไร? หาได้ยังไง?

Locator คือวิธีบอก Playwright ว่า element อยู่ที่ไหนบนหน้าเว็บ

วิธีหา Locator บน Chrome:
1. เปิดเว็บที่ต้องการ
2. คลิกขวาที่ element ที่ต้องการ
3. เลือก "Inspect"
4. ดู HTML แล้วหา id หรือ class ของ element นั้น
รูปแบบ Locatorใช้เมื่อตัวอย่าง
#user-nameหาด้วย idelement มี id="user-name"
.error-messageหาด้วย classelement มี class="error-message"
button[type='submit']หาด้วย attributeปุ่ม submit
text=Loginหาด้วยข้อความelement ที่มีคำว่า Login
[data-test='error']หาด้วย data attributeelement ที่มี data-test="error"

❓ คำถามสัมภาษณ์ที่ควรรู้

Q: Page Object Model คืออะไร?
A: Pattern แยก locator/action ออกเป็น class แยก ทำให้ maintain ง่าย แก้ที่เดียวทุก test ได้รับผล

Q: Locator คืออะไร?
A: วิธีบอก Playwright ว่า element อยู่ที่ไหนบนหน้าเว็บ เช่น #user-name, .error-message

Q: Playwright ต่างจาก Selenium ยังไง?
A: Auto-wait (รอ element พร้อมเอง), เร็วกว่า, API ใช้งานง่ายกว่า, มี Trace Viewer

Q: --headed กับ headless ต่างกันยังไง?
A: headed = เปิด browser จริงให้เห็น, headless = ไม่เปิด browser เร็วกว่า ใช้ใน CI/CD
